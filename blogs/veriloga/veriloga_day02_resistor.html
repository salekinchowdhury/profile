module simple_resistor(p, n);
    inout p, n;               // Analog ports
    electrical p, n;          // Declare these ports as electrical signals
  
    parameter real R = 1k;    // Parameter for resistance value

    analog begin
        V(p, n) <+ I(p, n) * R;  // Ohm's law: V = IR
    end
endmodule
